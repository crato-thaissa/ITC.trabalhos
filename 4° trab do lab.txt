#include <stdio.h>                                  
#include <conio.h>
#include <locale.h>
 
int fatorial(int x)                         
{
    int fat=0;                    
    for(fat = 1; x > 1; x--)
    {
        fat = x * fat;
    }
    return fat;                                     
}
int potencia(int base, int expoente)
{
	int cont, potencia=1;
	for(cont = 0; cont < expoente; cont++)
	{
		potencia = potencia * base;
	}
	return potencia;
}
int mdc(int x, int y)
{
	int r;
	while(y != 0){
		r = x % y;
		x = y;
		y = r;
	}
    return x;
}

	scanf("%d", &opc);
